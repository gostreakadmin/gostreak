Prepare the Google Sheet
Create a new Google Sheet.

In the first row (Row 1), enter these headers exactly:
name, email, leetcode_user, leetcode_pass, reference_id, transaction_id, language, contact, payment_screenshot

Step 2: Create the Apps Script
In your sheet, go to Extensions > Apps Script.

Paste this code (it saves text to the sheet and the image to your Drive):

JavaScript


function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var folder = DriveApp.getRootFolder(); // Images will save to your main Drive folder
    
    // 1. Process Image
    var data = Utilities.base64Decode(e.parameter.image_data);
    var blob = Utilities.newBlob(data, e.parameter.image_type, e.parameter.image_name);
    var file = folder.createFile(blob);
    var fileUrl = file.getUrl();

    // 2. Map Data to Sheet
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var nextRow = sheet.getLastRow() + 1;
    var newRow = [];

    for (var i = 0; i < headers.length; i++) {
      var header = headers[i];
      if (header === "payment_screenshot") {
        newRow.push(fileUrl);
      } else {
        newRow.push(e.parameter[header] || "");
      }
    }

    sheet.getRange(nextRow, 1, 1, newRow.length).setValues([newRow]);

    return ContentService.createTextOutput(JSON.stringify({"result":"success"}))
           .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"result":"error", "error": error.toString()}))
           .setMimeType(ContentService.MimeType.JSON);
  }
}
